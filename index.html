Useage 

下载源代码

引用样式文件Iva.css
<link rel="stylesheet" type="text/css" href="Iva.css">
引用脚本文件Iva.js
<script type="text/javascript" src="Iva.js"></script>
创建Iva实例
new Iva('playerCnt'/* 父容器id，为了保证浏览体验，请给父容器设置宽高860px*480px以上*/, {
  video: 'http://v.youku.com/v_show/id_XODIzNTE5NTY0.html',/* 视频播放地址 */
  cover: 'http://img0.bdstatic.com/img/image/4a75a05f8041bf84df4a4933667824811426747915.jpg'/* 视频封面，可选 */
});
