Usage 

      <link rel="stylesheet" type="text/css" href="Iva.css">
<script type="text/javascript" src="Iva.js"></script>

/**
 * 创建Iva实例
 */
var ivaInstance = new Iva('父容器',// 父容器id或者父容器DOM对象，给父容器设置宽高760px*480px以上可以获得最佳的浏览体验；如果宽高在760px*480px以下，Iva会以弹窗形式呈现
  {
    video: '地址',//必填，http://v.youku.com/v_show/id_XODIzNTE5NTY0.html
    type: 0,//可选，视频类型，0代表播放地址，1代表资源地址，默认为0
    title: '小视频',//可选，视频标题，如果type为1，该项为必填
    cover: '视频封面',//可选，视频封面
    logoPosition: 'left',//可选，自定义LOGO位置，“left”表示左上角，“right”表示右上角，默认为“left”
    autoplay: true//可选，是否自动播放，默认为true
  }
);

/**
 * 销毁Iva实例
 */
ivaInstance.destroy();
